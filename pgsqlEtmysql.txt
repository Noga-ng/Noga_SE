1. Materialized View

CREATE MATERIALIZED VIEW ventes_resume AS
SELECT
    client_id,
    SUM(total) AS montant
FROM ventes
GROUP BY client_id;

REFRESH MATERIALIZED VIEW ventes_resume;

RETURNING

INSERT INTO clients(nom)
VALUES ('Jean')
RETURNING id, nom;

3. DISTINCT ON
Retourne une seule ligne par client.

SELECT DISTINCT ON (client_id)
    client_id,
    date_commande,
    total
FROM commandes
ORDER BY client_id, date_commande DESC;

4. FILTER
SELECT
    COUNT(*) AS total,
    COUNT(*) FILTER (
        WHERE sexe='H'
    ) AS hommes,
    COUNT(*) FILTER (
        WHERE sexe='F'
    ) AS femmes
FROM personnes;


5. LATERAL
SELECT
    c.nom,
    x.total
FROM clients c

CROSS JOIN LATERAL (

    SELECT SUM(total) total

    FROM commandes

    WHERE client_id=c.id

) x;

8. Domain Type
CREATE DOMAIN email_type

AS TEXT

CHECK(

VALUE LIKE '%@%'

);

Puis

CREATE TABLE users(

email email_type

);


9. Range Type
CREATE TABLE reservation(

periode tsrange

);

Recherche

SELECT *

FROM reservation

WHERE

periode

&&

tsrange(

'2026-01-01',

'2026-01-10'

);


10. JSONB
SELECT *

FROM produits

WHERE

infos

@>

'{"marque":"Dell"}';

L'opérateur @> recherche un objet JSON contenu dans un autre.


11. GIN Index
CREATE INDEX idx_json

ON produits

USING GIN(infos);

Parfait pour les recherches dans JSONB.

12. GiST
CREATE INDEX idx_geo

ON villes

USING GIST(position);

Utilisé pour les données spatiales et certains types spécialisés.

13. BRIN
CREATE INDEX idx_date

ON ventes

USING BRIN(date);

Très efficace sur des tables gigantesques où les données sont naturellement ordonnées.
